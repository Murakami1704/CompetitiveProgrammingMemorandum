ABC281 C - Circular Playlist

AC灰

考えたことなど：
1曲目が流れ始めた時間をi曲目が流れ終わる時間を求める。この配列をTArrayとしたとき、TArray[i]はA[1]からA[i]までの累積和で求めることができる。全ての曲の秒数をTsumとしたとき、これはTArray[N]である。何曲目かを求めるためにT % Tsumを求め、これをSとする。SをTArrayに対して、二分探索を行うことで、S以上のイテレーターitが求められる。つまり、TArray[it] >= S。このitが何曲目かを表し、S - TArray[it-1]を求めることでその曲の開始からの時間も求められる。

View Code on AtCoder

コードを表示 
#include 
using namespace std;
int main() {
    long long n, t;
    cin >> n >> t;
    vector a(n+1, 0);
    for(int i = 0; i < n; i++){
        long long l;
        cin >> l;
        a[i+1] = l + a[i];
    }

    t %= a[n];
    int pos = lower_bound(a.begin(), a.end(), t) - a.begin();
    cout << pos << " " << t - a[pos-1] << endl;
    return 0;
}

ABC205 D - Kth Excluded

解説AC茶

考えたことなど：
こういう「いい数字」系列の問題、非常に苦手なんですわ。。。即座に解説を見て解きました。A1, A2, ..., AN のいずれとも異なる正整数をいい整数と呼ぶことにする。Aiまでのいい整数を保存する配列Ciを用意する。まず、Ki > ANの時は必ず答えはKi + Nとなる。問題はKi < ANの時であり、これはC1からCNについて、Kiを二分探索し、イテレーターitを取得する。この時、A[it] >= Ki、C[it] >= Kiである。この時、itは飛ばした数字の量と同義なので、Ki + itが答えとなる。

View Code on AtCoder

コードを表示 
#include 
using namespace std;
int main() {
    int n, q;
    cin >> n >> q;
    vector a(n), c(n);
    for(int i = 0; i < n; i++){
        cin >> a[i];
        if(i == 0){
            c[i] = a[i] - 1;
        }
        else {
            c[i] = c[i-1] + (a[i] - a[i-1] - 1);
        }
    }

    for(int i = 0; i < q; i++){
        long long k;
        cin >> k;
        if(c[n-1] < k){
            cout << k + n << endl;
        }
        else {
            int pos = lower_bound(c.begin(), c.end(), k) - c.begin();
            cout << k + pos << endl;
        }
    }
    return 0;
}

ABC321 D - Set Menu

解説AC緑

考えたこと：
Bの配列を昇順にソートし、Ai - Pという要素についてBで二分探索を行う。これで得られたイテレータitを境に、P * (M - it) + B[it]までの累積和 + Ai * itを0からNまで足し合わせることで求めることができる。

View Code on AtCoder

コードを表示 
#include 
using namespace std;
int main() {
    long long n, m, p;
    cin >> n >> m >> p;
    vector a(n), b(m), bs(m + 1, 0);
    for(int i = 0; i < n; i++){
        cin >> a[i];
    }
    for(int i = 0; i < m; i++){
        cin >> b[i];
    }
    sort(b.begin(), b.end());
    for(int i = 0; i < m; i++){
        bs[i+1] = bs[i] + b[i];
    }

    long long ans = 0;

    for(int i = 0; i < n; i++){
        long long t = p - a[i];
        int pos = lower_bound(b.begin(), b.end(), t) - b.begin();
        ans += bs[pos] + a[i] * pos;
        ans += (m - pos) * p;
    }

    cout << ans << endl;
    return 0;
}

ABC130 D - Enough Array

AC緑

考えたことなど：
まず、aの累積和を取る。そして、a[i]を起点に、a[it] - a[i] >= Kとなる最小のitを二分探索によって求める。N-itの総和が答えとなる。所謂尺取り法の解き方。

View Code on AtCoder

コードを表示 
#include 
using namespace std;
vector c(1000009);
int main() {
    int n;
    long long k;
    cin >> n >> k;
    vector a(n), s(n + 1, 0);
    for(int i = 0; i < n; i++){
        cin >> a[i];
        s[i+1] = s[i] + a[i];
    }
    long long count = 0;

    for(int l = 0; l <= n; l++){
        int r = lower_bound(s.begin(), s.end(), s[l] + k) - s.begin();
        if(r <= n){
            count += n - r + 1;
        }
    }
    
    cout << count << endl;
    return 0;
}

ABC106 D - AtCoder Express 2

解説AC水

考えたこと：
制約が500程度ということから、500*500の二次元配列Gを形成することができる。一次元目をl、二次元目をrとしたとき、まず、liとriを入力し、G[li][ri]++をする。そして、一次元目はlがnから0に向けて、二次元目はrが0からnに向けて累積和を取る。この時G[pi][qi]が答えとなる。

View Code on AtCoder

コードを表示 
#include 
using namespace std;
int main() {
    int n, m, q;
    cin >> n >> m >> q;
    vector> g(n, vector(n, 0));
    for(int i = 0; i < m; i++){
        int l, r;
        cin >> l >> r;
        g[l-1][r-1]++;
    }

    for(int i = n-2; i >= 0; i--){
        for(int j = 0; j < n; j++){
            g[i][j] += g[i+1][j];
        }
    }

    for(int j = 1; j < n; j++){
        for(int i = 0; i < n; i++){
            g[i][j] += g[i][j-1];
        }
    }

    for(int i = 0; i < q; i++){
        int l, r;
        cin >> l >> r;
        cout << g[l-1][r-1] << endl;
    }
    return 0;
}

ABC105 D - Candy Distribution

解説AC水

考えたこと：
やたらと難しかった気がする。。。Aiの累積和を取り、mで割った余りを求める。そして、このあまりについてmapにその数を保存する。最後にmapを全探索して、その組み合わせを答えに足していくことでACすることができる。

View Code on AtCoder

コードを表示 
#include 
using namespace std;
int main() {
    long long n, m;
    cin >> n >> m;
    vector a(n), s(n+1, 0);
    for(int i = 0; i < n; i++){
        cin >> a[i];
    }
    map mp;
    mp[0] = 1;
    for(int i = 1; i <= n; i++){
        s[i] = s[i-1] + a[i-1];
        mp[s[i] % m]++;
    }

    long long ans = 0;

    for (auto it = mp.begin(); it != mp.end(); ++it) {
        long long t = it->second;
        ans += t * (t - 1) / 2;
    }

    cout << ans << endl;
    return 0;
}

考えたこと：
n件についていもす法を実施し、その配列の最大値が答えになる。1,000,000という制約があったので間に合うか微妙だったけど、割と余裕がありました。

    int n;

        cin >> a[i] >> b[i];
        c[a[i]]++;
        c[b[i]+1]--;
    }

    ans = max(ans, c[0]);

        c[i] += c[i-1];
        ans = max(ans, c[i]);
    }

ABC017 C - ハイスコア

解説AC青

考えたことなど：
初めての青diffAC(解説だけど)。スコアの理論値から、宝石iを含む遺跡のスコアの差を取る。Scoresum - Scorei。これの最大値が答えとなる。問題はぱっと見難しそうだが、やっていることは余事象を取るみたいな考え方に近い。

コードを表示 
#include 
using namespace std;
int main() {
    int n, m;
    cin >> n >> m;
    vector g(m+2, 0);
    long long s = 0;
    for(int i = 0; i < n; i++){
        int l, r, sc;
        cin >> l >> r >> sc;
        s += sc;
        g[l] += sc;
        g[r+1] -= sc;
    }

    long long mi = (1 << 30);

    for(int i = 1; i < m+2; i++){
        g[i] += g[i-1];
    }

    for(int i = 1; i <= m; i++){
        mi = min(mi, g[i]);
    }

    cout << s - mi << endl;
    return 0;
}
