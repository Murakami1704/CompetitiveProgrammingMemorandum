ABC365 C - Transportation Expenses

AC灰

考えたことなど：
答え二分探索の教科書的な問題。$A_i$ の総和を求めて、まずこれが $M$ よりも小さかったら無限（infinite）、そうでなかった時に交通費補助額の上限額について二分探索を行うことで解ける。

View Code on AtCoder

コードを表示 
#include 
using namespace std;
int main() {
    long long n, m;
    cin >> n >> m;
    vector a(n);
    long long total = 0;
    for(int i = 0; i < n; i++){
        cin >> a[i];
        total += a[i];
    }

    if(total <= m){
        cout << "infinite" << endl;
        return 0;
    }

    long long low = 0, high = 2e14;
    while(abs(low - high) > 1){
        long long mid = (low + high) / 2;
        long long temp = 0;
        for(int i = 0; i < n; i++){
            temp += min(mid, a[i]);
        }

        if(temp <= m){
            low = mid;
        }
        else {
            high = mid;
        }
    }

    cout << low << endl;
    return 0;
}

ABC146 C - Buy an Integer

AC茶

考えたことなど：
これも答え二分探索の典型問題。$d(N)$ を求める方法は多岐にわたるが、今回は10で割っていって桁数を求める方法を採用。あとは予算内で買うことのできる最も大きい整数について二分探索することで答えに導ける。

View Code on AtCoder

コードを表示 
#include 
using namespace std;
int main() {
    long long a, b, x;
    cin >> a >> b >> x;

    long long low = 0, high = 1e9 + 1;
    while(high - low > 1){
        long long mid = (low + high) / 2;
        long long keta = 0, temp = mid;
        while(temp != 0){
            temp /= 10;
            keta++;
        }

        if(a * mid + b * keta <= x){
            low = mid;
        }
        else {
            high = mid;
        }
    }

    cout << low << endl;
    return 0;
}

ABC319 D - Minimum Width

解説AC茶

考えたことなど：
答え二分探索の典型。判定部分の貪欲法で手間取ってしまい、4回ほどWAを出してしまった。基本的な処理こそ丁寧に行う必要があると痛感。

View Code on AtCoder

コードを表示 
#include 
using namespace std;
int main() {
    int n, m;
    cin >> n >> m;
    vector a(n);
    long long s = 0, ma = 0;;
    for(int i = 0; i < n; i++){
        cin >> a[i];
        s += a[i];
        ma = max(ma, a[i]);
    }

    long long low = ma - 1, high = s + n - 1;
    while(high - low > 1){
        long long mid = (low + high) / 2;

        int row = 1;
        long long l = 0;
        for(int i = 0; i < n; i++){
            if(l == 0){
                l = a[i];
            }
            else if(l + 1 + a[i] <= mid){
                l += 1 + a[i];
            }
            else{
                row++;
                l = a[i];
            }
        }

        if(row > m){
            low = mid;
        }
        else {
            high = mid;
        }
    }

    cout << high << endl;
    return 0;
}

ABC174 E - Logs

解説AC水

考えたことなど：
一見どうすればいいかわからない、少しひねった問題に見えるが、丸太の「長さ」について二分探索すればよい。各丸太を目標の長さ以下にするために必要なカット回数を合計し、それが $K$ 回以内か判定する。

View Code on AtCoder

コードを表示 
#include 
using namespace std;
int main() {
    long long n, k, m;
    cin >> n >> k;
    vector a(n);
    for(int i = 0; i < n; i++){
        cin >> a[i];
        m = max(m, a[i]);
    }

    long long low = 0, high = m;
    while(high - low > 1){
        long long mid = (high + low) / 2;
        long long count = 0;
        for(int i = 0; i < n; i++){
            count += (a[i] - 1) / mid;
        }

        if(count <= k){
            high = mid;
        }
        else {
            low = mid;
        }
    }

    cout << high << endl;
    return 0;
}

ABC227 D - Project Planning

解説AC青

考えたことなど：
答え二分探索の雰囲気を感じることはできたが、判定式が思い浮かばず解説を読了。初見でこの判定ロジックを思いつくにはまだ修行が必要。また、この問題で __int128 の存在を初めて知った。long long を超える範囲を扱える武器が増えたのは大きい。

View Code on AtCoder

コードを表示 
#include 
using namespace std;
int main() {
    long long n, k;
    cin >> n >> k;
    vector a(n);
    for(int i = 0; i < n; i++){
        cin >> a[i];
    }

    long long low = 0, high = 1e18;
    while(high - low > 1){
        long long mid = (low + high) / 2;
        __int128 sum = 0;
        for(int i = 0; i < n; i++){
            sum += min(a[i], mid);
        }
        if(sum >= (__int128)mid * k){
            low = mid;
        }
        else {
            high = mid;
        }
    }

    cout << low << endl;
    return 0;
}

ABC457 D - Raise Minimum

AC茶

考えたことなど：
先週のABCで歯が立たず、宿敵としていた問題をついに撃破！答え二分探索を知っていれば比較的余裕を持って解ける典型的な問題だった。ただし、計算の過程で long long を超えるため __int128 で記述。出力時には long long にキャストするなどの工夫が必要だった。

View Code on AtCoder

コードを表示 
#include 
using namespace std;
int main() {
    long long n, k;
    cin >> n >> k;
    vector a(n);
    for(int i = 0; i < n; i++){
        cin >> a[i];
    }

    __int128 low = 0, high = 1e30;
    while(high - low > 1){
        long long mid = (low + high) / 2;
        
        __int128 count = 0;
        for(long long i = 0; i < n; i++){
            if(mid >= a[i]){
                count += (mid - a[i] + i) / (i + 1);
            }
        }

        if(count > k){
            high = mid;
        }
        else {
            low = mid;
        }
    }

    cout << (long long)low << endl;
    return 0;
}
